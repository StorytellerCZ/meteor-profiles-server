    type: String,
    optional: true

Meteor.methods({
  /**
   * Updates user's avatar
   * @param {string} avatar
   * @returns {boolean}
   */
  updateAvatar:function(avatar){
    check(avatar, String)
    let profile = Meteor.profiles.findOne({userId: Meteor.userId()}).update({$set: {avatar: avatar}})
    if(profile === undefined){
      return true
    } else {
      return false
    }
  },
  /**
